CREATE TABLE "board_question" ("id" integer NOT NULL PRIMARY KEY AUTOINCREMENT, "title" varchar(200) NOT NULL, "content" text NOT NULL, "create_date" datetime NOT NULL, "modify_date" datetime NOT NULL, "views" integer NOT NULL, "imgs" varchar(100) NULL, "author_id" varchar(100) NULL REFERENCES "common_user" ("nickname") ON UPDATE CASCADE DEFERRABLE INITIALLY DEFERRED, "college_id" varchar(50) NULL REFERENCES "common_college" ("department") DEFERRABLE INITIALLY DEFERRED)
